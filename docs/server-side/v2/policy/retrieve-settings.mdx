This endpoint is used to fetch settings.

<GetRequest title="https://api.100ms.live/v2/templates/<template_id>/settings" />

<Tabs items={['Request', 'Response']} id="req" />

<Tab id="req-0">

```bash
curl --location --request GET 'https://api.100ms.live/v2/templates/<template_id>/settings' \
--header 'Authorization: Bearer <management_token>'
```

</Tab>

<Tab id="req-1">
<Response res="200: OK" />

```json
{
    "region": "in",
    "subscribeDegradation": {},
    "screenSimulcastLayers": {},
    "videoSimulcastLayers": {},
    "recording": null,
    "roomState": {
        "messageInterval": 5,
        "sendPeerList": false,
        "stopRoomStateOnJoin": true,
        "enabled": false
    },
    "retry": {}
}
```

</Tab>
