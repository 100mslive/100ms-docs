Trigger this request to stop rtmp streaming and / or recording.

<PostRequest title="https://api.100ms.live/v2/external-streams/:job_id/stop" />
<Request id="req-0">

```bash
curl --location --request POST 'https://api.100ms.live/v2/external-streams/<job_id>/stop' \
--header 'Authorization: Bearer <management_token>' \
--header 'Content-Type: application/json' \
--data-raw '{
    "meeting_url" : "<meeting_url>",
    "rtmp_urls": [<rtmp_url_1>, <rtmp_url_2>],
    "record": true,
    "resolution" : {"width": 1280, "height": 720},
    "destination": "<destination_name>"
}'
```

</Request>
